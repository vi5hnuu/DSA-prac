class Solution
{
    public:
    int orangesRot(vector<vector<int>>grid){
	    int rows=grid.size();
	    int cols=grid[0].size();

	    queue<pair<int,int>> que;
	    int freshCount=0;
	    for(int i=0;i<rows;i++){
	        for(int j=0;j<cols;j++){
	            if(grid[i][j]==2){
	                que.push({i,j});
	            }
	            if(grid[i][j]==1){
	                freshCount++;
	            }
	        }
	    }
	    if(!freshCount) return 0;
	    //for all initial ones there is zero in distacnce matrix already
	    int step=1;
	    while(!que.empty()){
	        int sz=que.size();
	        while(sz>0){
	            sz--;
	            auto pos=que.front();
	            que.pop();
	            int ai[4]={0,0,1,-1};
	            int aj[4]={1,-1,0,0};
	            for(int j=0;j<4;j++){
	                int x=pos.first+ai[j];
	                int y=pos.second+aj[j];
	                if(x>=0 && x<rows && y>=0 && y<cols && grid[x][y]==1){
    	                freshCount--;
    	                grid[x][y]=2;
    	                que.push({x,y});
	                }
	            }
	        }
	        if(!freshCount) break;
	        step++;
	    }
	    return freshCount ? -1 : step;
	}
};
