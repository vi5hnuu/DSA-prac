```cpp
int find(int par[], int x) {
    if(par[x]==x) return x;
    //compression [flatten]
    return par[x]=find(par,par[x]);
}

void unionSet(int par[], int x, int z) {
    int rx=find(par,x);
    int rz=find(par,z);
    if(rx==rz) return;

    par[rx]=rz;
}
```
