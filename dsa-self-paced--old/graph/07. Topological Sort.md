```cpp
class Solution {
  public:
    vector<int> topoSort(int V, vector<vector<int>>& edges) {
        vector<vector<int>> adj(V,vector<int>());

        vector<int> indegree(V,0);
        for(auto &edge:edges){
            adj[edge[0]].push_back(edge[1]);
            indegree[edge[1]]++;
        }

        queue<int> nodes;
        for(int i=0;i<V;i++){
            if(!indegree[i]) nodes.push(i);
        }

        vector<int> topo;
        int indCount=nodes.size();
        while(!nodes.empty()){
            int node=nodes.front();
            nodes.pop();
            topo.push_back(node);
            for(int neighbour:adj[node]){
                indegree[neighbour]--;
                if(!indegree[neighbour]) {
                    nodes.push(neighbour);
                    indCount++;
                }
            }
        }
        if(indCount==V) return topo;
        return {};
    }
};
```

```cpp
class Solution {
  public:
    vector<int> topoSort(int V, vector<vector<int>>& edges) {
        vector<vector<int>> adj(V,vector<int>());
        for(auto &edge:edges){
            adj[edge[0]].push_back(edge[1]);
        }
        vector<int> topo;
        vector<bool> visited(V,false);

        for(int i=0;i<V;i++){
            if(visited[i]) continue;
            fillTopo(adj,visited,topo,i);
        }
        return vector<int>(topo.rbegin(),topo.rend());
    }
  private:
    void fillTopo(vector<vector<int>> &adj,vector<bool> &visited,vector<int> &topo,int root){
        if(visited[root]) return;

        visited[root]=true;

        for(int neighbour:adj[root]){
            if(visited[neighbour]) continue;
            fillTopo(adj,visited,topo,neighbour);
        }
        topo.push_back(root);
    }
};
```
