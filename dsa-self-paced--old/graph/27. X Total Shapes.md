```cpp
class Solution {
  public:
    // Function to find the number of 'X' total shapes.
    int xShape(vector<vector<char>>& grid) {
        int r=grid.size();
        int c=grid[0].size();

        int xCount=0;
        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(grid[i][j]!='X') continue;
                dfs(r,c,grid,i,j);
                xCount++;
            }
        }
        return xCount;
    }
  private:
    void dfs(int r,int c,vector<vector<char>> &grid,int i,int j){
        if(i<0 || j<0 || i>=r || j>=c || grid[i][j]!='X') return;

        grid[i][j]='O';

        int dx[]={1,-1,0,0};
        int dy[]={0,0,-1,1};
        for(int k=0;k<4;k++){
            int x=i+dx[k];
            int y=j+dy[k];
            dfs(r,c,grid,x,y);
        }
    }
};
```

```cpp


class Solution {
  public:
    // Function to find the number of 'X' total shapes.
    int xShape(vector<vector<char>>& grid) {
        int r=grid.size();
        int c=grid[0].size();

        int xCount=0;
        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(grid[i][j]!='X') continue;
                dfs(r,c,grid,i,j);
                xCount++;
            }
        }
        return xCount;
    }
  private:
    void dfs(int r,int c,vector<vector<char>> &grid,int i,int j){
        if(i<0 || j<0 || i>=r || j>=c || grid[i][j]!='X') return;

        queue<pair<int,int>> nodes;
        nodes.push({i,j});
        grid[i][j]='O';

        int dx[]={1,-1,0,0};
        int dy[]={0,0,-1,1};
        while(!nodes.empty()){
            int sz=nodes.size();

            while(sz--){
                auto [i,j]=nodes.front();
                nodes.pop();

                for(int k=0;k<4;k++){
                    int x=i+dx[k];
                    int y=j+dy[k];
                    if(x>=0 && y>=0 && x<r && y<c && grid[x][y]=='X'){
                        grid[x][y]='O';
                        nodes.push({x,y});
                    }
                }
            }
        }
    }
};
```
