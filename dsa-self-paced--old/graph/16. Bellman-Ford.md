// User function Template for C++

class Solution {
  public:
    vector<int> bellmanFord(int V, vector<vector<int>>& edges, int src) {
        vector<int> distance(V,INT_MAX);
        distance[src]=0;

        for(int i=0;i<V-1;i++){//because a path of v vertexs has v-1 edges
            for(auto &edge:edges){
                int u=edge[0];
                int v=edge[1];
                int w=edge[2];
                if(distance[u]==INT_MAX) continue;
                distance[v]=min(distance[u]+w,distance[v]);
            }
        }
        for(auto &edge:edges){
                int u=edge[0];
                int v=edge[1];
                int w=edge[2];
                if(distance[u]==INT_MAX) continue;
                if(distance[u]+w<distance[v]) return {-1};
        }
        for(int i=0;i<V;i++){
            if(distance[i]==INT_MAX) distance[i]=100000000;
        }
        return distance;
    }
};
