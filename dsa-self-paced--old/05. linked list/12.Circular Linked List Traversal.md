```cpp
/* structure for a node
struct Node
{
    int data;
    struct Node *next;
}; */

/* Function to print nodes in a given Circular linked list */
void printList(struct Node *head) {
    if(!head) return;

    Node *nH=head;
    while(nH){
        cout<<nH->data<<" ";
        if(nH->next==head) break;
        nH=nH->next;
    }

}
```
