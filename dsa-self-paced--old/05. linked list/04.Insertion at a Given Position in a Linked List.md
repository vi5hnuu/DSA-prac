```cpp
/*
class Node {
  public:
    int data;
    Node *next;
    Node(int x) {
        data = x;
        next = nullptr;
    }
};
*/

class Solution {
  public:
    Node *insertPos(Node *head, int pos, int val) {
        Node *node=new Node(val);
        if(pos<=1){
            node->next=head;
            return node;
        }

        Node *nH=head;
        while(pos-2>0){
            if(!nH) throw "invalid position";
            nH=nH->next;
            pos--;
        }

        node->next=nH->next;
        nH->next=node;

        return head;
    }
};
```
