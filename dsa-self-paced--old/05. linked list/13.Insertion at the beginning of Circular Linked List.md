```cpp
/*
class Node {
  public:
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/
class Solution {
  public:
    Node* insertAtBeginning(Node* tail, int key) {
        Node* node=new Node(key);
        if(!tail){
            node->next=node;
            return node;
        }
        node->next=tail->next;
        tail->next=node;
        return node;
    }
};
```
