```cpp
/*
class Node {
  public:
    int data;
    Node* next;
    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    Node* addTwoLists(Node* head1, Node* head2) {
        while(head1 && !head1->data){
            // Node *toDel=head1;
            head1=head1->next;
            // delete toDel;
        }
        while(head2 && !head2->data){
            // Node *toDel=head2;
            head2=head2->next;
            // delete toDel;
        }
        Node *n1=revLL(head1);
        Node *n2=revLL(head2);

        Node *sm=nullptr;
        int carry=0;
        while(n1 || n2 || carry){
            int total=(n1 ? n1->data : 0) + (n2 ? n2->data : 0) + carry;
            Node *digit=new Node(total%10);
            carry=total/10;
            digit->next=sm;
            sm=digit;
            if(n1) n1=n1->next;
            if(n2) n2=n2->next;
        }
        return sm;
    }
  private:
    Node* revLL(Node *head){
        Node *rev=nullptr;

        while(head){
            Node *nxt=head->next;
            head->next=rev;
            rev=head;
            head=nxt;
        }
        return rev;
    }
};
```
