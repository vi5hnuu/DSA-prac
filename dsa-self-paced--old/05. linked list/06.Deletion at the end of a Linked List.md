```cpp
/*
class Node {
public:
    int data;
    Node* next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/
class Solution {
  public:
    Node* removeLastNode(Node* head) {
        if(!head) return head;
        if(!head->next){
            delete head;
            return nullptr;
        }
        Node *nH=head;
        while(nH && nH->next && nH->next->next) nH=nH->next;
        if(!nH) return head;
        if(nH->next) delete nH->next;
        nH->next=nullptr;
        return head;
    }
};
```
