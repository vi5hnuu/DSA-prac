```cpp
// User function template for C++

/*
struct Node
{
    int data;
    struct Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }
};
*/
class Solution {
  public:
    Node* divide(Node* head) {
        Node *even=nullptr,*evenT=nullptr;
        Node *odd=nullptr,*oddT=nullptr;

        while(head){
            if(head->data%2==0){
                if(!even) even=evenT=head;
                else evenT=evenT->next=head;
            }else{
                if(!odd) odd=oddT=head;
                else oddT=oddT->next=head;
            }
            head=head->next;
        }

        if(!even) return odd;

        evenT->next=odd;
        if(oddT) oddT->next=nullptr;
        return even;
    }
};
```
