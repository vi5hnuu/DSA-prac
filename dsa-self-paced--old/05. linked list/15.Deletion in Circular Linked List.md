```cpp
/*
class Node {
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/
class Solution {
  public:
    Node* deleteNode(Node* head, int key) {
        if(!head) return nullptr;
        if(head->data==key && head->next==head){
            delete head;
            return nullptr;
        }

        //if first node is key...change starting node
        if(head->data==key) head=head->next;

        Node *nH=head;
        while(nH){
            if(nH->next->data==key) break;
            if(nH->next==head) return head;
            nH=nH->next;
        }

        Node *toDel=nH->next;
        nH->next=toDel->next;
        delete toDel;
        return head;
    }
};
```
