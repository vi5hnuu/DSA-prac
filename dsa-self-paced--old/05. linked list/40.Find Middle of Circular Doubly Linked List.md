```cpp
/*
class Node
{
  public:
    int data;
    Node * next;
    Node * prev;
    Node (int x)
    {
        data=x;
        next=NULL;
        prev=NULL;
    }
};
*/

class Solution {
  public:
    int findMiddle(Node* head) {//for odd length
        Node *slow=head,*fast=head;

        while(fast && fast->next){
            slow=slow->next;
            fast=fast->next->next;
            if(fast->next==head) break;
        }
        return slow->data;
    }
};
```
