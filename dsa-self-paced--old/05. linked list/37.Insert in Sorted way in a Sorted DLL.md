/*structure of the node of the DLL is as
struct Node {
    int data;
    struct Node* prev, *next;
};
*/

// function should insert a new node in sorted way in
// a sorted doubly linked list
// Return the head after insertion
class Solution {
  public:
    Node* sortedInsert(Node* head, int x) {
        Node *node=new Node();
        node->data=x;

        if(!head) return node;

        if(head->data>=x){
            node->next=head;
            head->prev=node;
            return node;
        }

        Node *nH=head;
        Node *prev=nullptr;
        while(nH && nH->data<x){
            prev=nH;
            nH=nH->next;
        }
        node->next=prev->next;
        if(prev->next) prev->next->prev=node;
        prev->next=node;
        node->prev=prev;

        return head;
    }
};
