```cpp
/*
class Node {
public:
    int data;
    Node* next;
    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    Node* joinTheLists(Node* head1, Node* head2) {
       if(!head1 || !head2) return head1 ? head1 : head2;

       Node *joined=head1;

       while(head1 && head1->next) head1=head1->next;
       head1->next=head2;
       return joined;
    }
};
```
