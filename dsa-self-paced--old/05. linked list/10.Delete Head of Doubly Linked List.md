// User function Template for C++

/*
struct Node
{
    int data;
    Node * next;
    Node * prev;
    Node (int x)
    {
        data=x;
        next=NULL;
        prev=NULL;
    }

};
*/
Node *deleteHead(Node *head) {
    if(!head) return nullptr;
    Node *toDel=head;
    head=head->next;
    if(head) head->prev=nullptr;
    delete toDel;
    return head;
}
