```cpp
/*
class Node {
 public:
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    int lengthOfLoop(Node *head) {
        Node *slow=head,*fast=head;

        while(fast && fast->next){
            slow=slow->next;
            fast=fast->next->next;
            if(slow==fast) break;
        }

        if(!fast || !fast->next) return 0;

        int loopLen=1;
        slow=slow->next;
        while(slow!=fast){
            loopLen++;
            slow=slow->next;
        };

        return slow!=fast ? 0 : loopLen;
    }
};
```
