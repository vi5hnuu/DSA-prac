// User function Template for C++

void rollingHash(string s, string p) {
    int slen=s.length();
    int plen=p.length();
    if(slen<plen) {
        cout<<-1<<endl;
        return;
    }

    int pHash=0;
    for(int i=0;i<plen;i++){
        pHash+=(p[i]-'a'+1);
    }

    int rollingHash=0;
    bool foundAny=false;
    for(int i=0;i<slen;i++){
        rollingHash+=(s[i]-'a'+1);
        if(i>=plen) {
            rollingHash-=(s[i-plen]-'a'+1);
        }
        if(i>=plen-1 && rollingHash==pHash) {
            cout<<s.substr(i-plen+1,plen)<<" "<<(i-plen+1)<<endl;
            foundAny=true;
        }
    }
    if(!foundAny){
        cout<<-1<<endl;
    }
}
