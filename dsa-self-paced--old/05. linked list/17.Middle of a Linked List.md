```cpp
/*
class Node {
    int data;
    Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }

}; */

class Solution {
  public:
    int getMiddle(Node* head) {
        int len=lenLL(head);
        for(int i=0;i<len/2;i++){
            head=head->next;
        }
        return head->data;
    }
  private:
    int lenLL(Node* node){
        int len=0;
        while(node){
            node=node->next;
            len++;
        }
        return len;
    }
};
```

```cpp
/*
class Node {
    int data;
    Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }

}; */

class Solution {
  public:
    int getMiddle(Node* head) {
        if(!head || !head->next) return head->data;
        Node* slow=head,*fast=head;

        while(slow && fast && fast->next){
            slow=slow->next;
            fast=fast->next->next;
        }
        return slow->data;
    }
};
```
