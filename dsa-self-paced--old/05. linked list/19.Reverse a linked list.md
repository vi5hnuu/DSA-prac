```cpp
/*
class Node {
 public:
    int data ;
    Node *next ;

    Node(int x) {
        data = x ;
        next = NULL ;
    }
};
*/

class Solution {
  public:
    Node* reverseList(Node* head) {
        Node *rev=nullptr;

        while(head){
            Node *nxt=head->next;
            head->next=rev;
            rev=head;
            head=nxt;
        }
        return rev;
    }
};
```

```cpp
/*
class Node {
 public:
    int data ;
    Node *next ;

    Node(int x) {
        data = x ;
        next = NULL ;
    }
};
*/

class Solution {
  public:
    Node* reverseList(Node* head) {
        if(!head || !head->next) return head;
        Node *hd=reverseList(head->next);//return always the last node

        //reverse arrow
        head->next->next=head;
        head->next=nullptr;

        return hd;//return always last node
    }
};
```

```cpp
/*
class Node {
 public:
    int data ;
    Node *next ;

    Node(int x) {
        data = x ;
        next = NULL ;
    }
};
*/

class Solution {
  public:
    Node* reverseList(Node* curr,Node *prev=nullptr) {
        if(!curr) return prev;

        Node *nxt=curr->next;
        curr->next=prev;


        return reverseList(nxt,curr);
    }
};
```
