```cpp
/* Structure of Node
class Node {
  public:
    int data;
    Node *next;
    Node *prev;

    Node(int val) {
        data = val;
        this->next = NULL;
        this->prev = NULL;
    }
};
*/

class Solution {
  public:
    Node* delPos(Node* head, int x) {
       x--;
       if(x<=0){
        Node *nxt=head->next;
        delete head;
        if(nxt) nxt->prev=nullptr;
        return nxt;
       }

       Node *nH=head;
       while(x>0 && nH){
           nH=nH->next;
           x--;
       }

       if(!nH) return head;

       if(nH->next) nH->next->prev=nH->prev;
       nH->prev->next=nH->next;
       delete nH;
       return head;
    }
};
```
