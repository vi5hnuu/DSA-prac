```cpp
int isCircular(Node* head) {
    if(!head) return true;
    Node *nH=head;

    while(nH){
        if(nH->next==head) return true;
        nH=nH->next;
    }
    return false;
}
```

```cpp
int isCircular(Node* head) {
    if(!head) return true;
    Node *slow=head,*fast=head;

    while(fast && fast->next){
        slow=slow->next;
        fast=fast->next->next;
        if(slow==fast) return true;
    }

    return false;
}
```
