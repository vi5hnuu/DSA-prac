```cpp
/*
structure of the node of the list is as
struct Node
{
    int data;
    struct Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    // Should return head of the modified linked list
    Node* sortedInsert(Node* head, int key) {
        Node *node=new Node(key);
        if(!head) return node;

        Node *prev=nullptr,*curr=head;

        while(curr && curr->data<key){
            prev=curr;
            curr=curr->next;
        }


        node->next=curr;
        if(!prev){
            node->next=curr;
            return node;
        }else{
            prev->next=node;
            return head;
        }
    }
};
```

```cpp
/*
structure of the node of the list is as
struct Node
{
    int data;
    struct Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    // Should return head of the modified linked list
    Node* sortedInsert(Node* head, int key) {
        Node *node=new Node(key);
        if(!head) return node;

        if(key<head->data){
            node->next=head;
            return node;
        }

        Node *curr=head;

        while(curr && curr->next && curr->next->data<key){
            curr=curr->next;
        }

        node->next=curr->next;
        curr->next=node;
        return head;
    }
};
```
