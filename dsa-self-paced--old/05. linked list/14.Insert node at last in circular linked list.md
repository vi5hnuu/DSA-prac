/*// creating struct Node
struct Node
{
    int data;
    Node *next;

};*/
Node* insert_at_end(Node* head, int x) {
    Node* node=new Node();
    node->data=x;

    if(!head){
        node->next=node;
        return node;
    }

    Node *nH=head;
    while(nH && nH->next!=head){
        nH=nH->next;
    }

    node->next=nH->next;
    nH->next=node;
    return head;
}
