```cpp
/*
class Node {
  public:
    int data;
    Node* next;

    // Default constructor
    Node() {
        data = 0;
        next = NULL;
    }

    // Parameterised Constructor
    Node(int data) {
        this->data = data;
        this->next = NULL;
    }
};
*/
class Solution {
  public:
    Node *insertAtEnd(Node *head, int x) {
        Node *node=new Node(x);
        if(!head) return node;

        Node *nH=head;
        while(nH->next) nH=nH->next;
        nH->next=node;
        return head;
    }
};
```
