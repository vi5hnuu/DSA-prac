```cpp
/*
class Node {
  public:
    int data;
    Node *next;
    Node *prev;
    Node(int val) {
        data = val;
        next = NULL;
        prev = NULL;
    }
};

*/
class Solution {
  public:
    Node *reverse(Node *head) {
        if(!head) return nullptr;
        Node *rev=head;
        head=head->next;
        rev->next=nullptr;

        while(head){
            Node *nxt=head->next;
            head->next=rev;
            rev=rev->prev=head;
            head=nxt;
        }
        return rev;
    }
};
```

```cpp
/*
class Node {
  public:
    int data;
    Node *next;
    Node *prev;
    Node(int val) {
        data = val;
        next = NULL;
        prev = NULL;
    }
};

*/
class Solution {
  public:
    Node *reverse(Node *head) {
        if(!head || !head->next) return head;

        Node *prev=nullptr,*curr=head;

        while(curr){
            prev=curr->prev;

            curr->prev=curr->next;
            curr->next=prev;
            curr=curr->prev;
        }
        return prev->prev;
    }
};
```
