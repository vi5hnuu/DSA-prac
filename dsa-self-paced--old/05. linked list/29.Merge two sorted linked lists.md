```cpp
/*
class Node {
 public:
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    Node* sortedMerge(Node* head1, Node* head2) {
        Node *s=nullptr,*st=nullptr;

        while(head1 || head2){
            Node *n=(head1 && head2) ? (head1->data<head2->data ? head1 : head2) : (head1 ? head1 : head2);
            if(n==head1) head1=head1->next;
            else head2=head2->next;
            if(!s) s=st=n;
            else st=st->next=n;
        }
        return s;
    }
};
```
