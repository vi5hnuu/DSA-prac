```cpp
/*
  Pairwise swap a linked list
  The input list will have at least one element
  node is defined as

struct Node
{
    int data;
    struct Node* next;

    Node(int x){
        data = x;
        next = NULL;
    }

}*head;
*/
class Solution {
  public:
    Node* pairWiseSwap(Node* head) {
        if(!head || !head->next) return head;
        Node *prevSwapTail=nullptr;
        Node *nH=nullptr;
        while(head && head->next){//swap two nodes per iteration
            Node *nxt=head->next;
            Node *nnxt=nxt ? nxt->next:nullptr;

            nxt->next=head;
            head->next=nnxt;
            if(prevSwapTail) prevSwapTail->next=nxt;
            else nH=nxt;
            prevSwapTail=head;
            head=nnxt;
        }
        return nH;
    }
};
```
