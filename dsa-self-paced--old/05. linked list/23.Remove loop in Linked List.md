```cpp
/*
class Node {
  public:
    int data;
    Node* next;

    Node(int val) {
        data = val;
        next = nullptr;
    }
};
*/
class Solution {
  public:
    void removeLoop(Node* head) {
        unordered_map<Node*,bool> visited;

        while(head){
            if(!head->next) return;
            if(visited[head->next]){
                head->next=nullptr;
                return;
            }
            visited[head]=true;
            head=head->next;
        }
    }
};
```

```cpp
/*
class Node {
  public:
    int data;
    Node* next;

    Node(int val) {
        data = val;
        next = nullptr;
    }
};
*/
class Solution {
  public:
    void removeLoop(Node* head) {
        if(!head) return;
        Node *slow=head,*fast=head;

        while(slow && fast && fast->next){
            slow=slow->next;
            fast=fast->next->next;

            if(slow==fast) break;
        }
        if(slow!=fast) return;

        slow=head;
        while(slow!=fast){
            slow=slow->next;
            fast=fast->next;
        }

        while(fast->next!=slow) fast=fast->next;
        fast->next=nullptr;
    }
};
```
