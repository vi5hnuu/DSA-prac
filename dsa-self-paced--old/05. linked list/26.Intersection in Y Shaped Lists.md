class Node {
public:
    int data;
    Node* next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    Node* intersectPoint(Node* head1, Node* head2) {
       unordered_map<Node*,bool> visited;

       while(head1){
           visited[head1]=true;
           head1=head1->next;
       }

       while(head2){
           if(visited[head2]) return head2;
           head2=head2->next;
       }
       return nullptr;
    }
};

class Solution {
  public:
    Node* intersectPoint(Node* head1, Node* head2) {
       if(!head1 || !head2) return head1 ? head1 : head2;

       int l1=llLen(head1);
       int l2=llLen(head2);

       int diff=abs(l1-l2);
       while(diff>0){
           if(l1<l2) head2=head2->next;
           else head1=head1->next;
           diff--;
       }

       while(head1 && head2 && head1!=head2){
           head1=head1->next;
           head2=head2->next;
       }
       return head1;
    }
  private:
    int llLen(Node *ll){
        int len=0;
        while(ll){
            len++;
            ll=ll->next;
        }
        return len;
    }
};

class Solution {
  public:
    Node* intersectPoint(Node* head1, Node* head2) {
       if(!head1 || !head2) return head1 ? head1 : head2;

       Node *x=head1,*y=head2;
       while(x && y && x!=y){
           x=x->next;
           y=y->next;
       }

       if(!x) x=head2;
       else if(!y) y=head1;

        while(x && y && x!=y){
           x=x->next;
           y=y->next;
       }

       if(!x) x=head2;
       else if(!y) y=head1;

       while(x && y && x!=y){
           x=x->next;
           y=y->next;
       }

       return x && y && x==y ? x:nullptr;
    }
};

class Solution {
  public:

       if(!head1 || !head2) return head1 ? head1 : head2;

       Node *x=head1,*y=head2;
       while(x != y){
           x=x->next;
           y=y->next;

           if(!x && !y) return nullptr;
           if(!x) x=head2;
           else if(!y) y=head1;
       }
       return x;
    }
};
