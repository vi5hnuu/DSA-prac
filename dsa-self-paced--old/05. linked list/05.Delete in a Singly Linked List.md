```cpp
/*
class Node {
public:
    int data;
    Node* next;

    Node(int data) {
        this->data = data;
        this->next = nullptr;
    }
};
*/
class Solution {
  public:
    Node* deleteNode(Node* head, int x) {
        Node *nH=head;
        if(x<=1){
          head=head->next;
          delete nH;
          return head;
        }
        while(x-2>0 && nH){
            nH=nH->next;
            x--;
        }
        if(!nH) return head;

        Node *toDel=nH->next;
        if(!toDel) return head;
        nH->next=toDel->next;
        delete toDel;
        return head;
    }
};
```
