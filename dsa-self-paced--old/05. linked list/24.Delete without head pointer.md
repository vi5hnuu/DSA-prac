https://www.geeksforgeeks.org/problems/delete-without-head-pointer/1

```cpp
/*
struct Node {
  int data;
  struct Node *next;
  Node(int x) {
    data = x;
    next = NULL;
  }
}*head;
*/

class Solution {
  public:
    void deleteNode(Node* del_node) {
        Node *nxt=del_node->next;
        *del_node=*nxt;
        delete nxt;
    }
};
```
