```cpp
/*
struct Node {
  int data;
  struct Node *next;
  Node(int x) {
    data = x;
    next = NULL;
  }
};*/

class Solution {
  public:
    // Function to remove duplicates from sorted linked list.
    Node* removeDuplicates(Node* head) {
        if(!head) return nullptr;
        Node *nH=head;
        while(nH){
            while(nH && nH->next && nH->next->data==nH->data){
                Node *toDel=nH->next;
                nH->next=toDel->next;
                delete toDel;
            }
            nH=nH->next;
        }
        return head;
    }
};
```
