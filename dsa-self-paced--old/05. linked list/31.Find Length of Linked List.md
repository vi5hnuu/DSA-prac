```cpp
/*
class Node {
  public:
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    int getCount(Node* head) {
        int len=0;
        while(head){
            head=head->next;
            len++;
        }
        return len;
    }
};
```

```cpp
/*
class Node {
  public:
    int data;
    Node *next;

    Node(int x) {
        data = x;
        next = NULL;
    }
};
*/

class Solution {
  public:
    int getCount(Node* head) {
        if(!head) return 0;
        return 1+getCount(head->next);
    }
};
```
