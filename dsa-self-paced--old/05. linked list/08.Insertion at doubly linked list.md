```cpp
/*
class Node {
  public:
    int data;
    Node* next;
    Node* prev;

    Node(int x) {
        data = x;
        next = prev = nullptr;
    }
};
*/

class Solution {
  public:
    Node *insertAtPos(Node *head, int p, int x) {
        Node *node=new Node(x);

        Node *nH=head;
        while(p>0) {
            nH=nH->next;
            p--;
        }
        if(!nH) return head;

        node->next=nH->next;
        if(nH->next) nH->next->prev=node;
        node->prev=nH;
        nH->next=node;
        return head;
    }
};
```
