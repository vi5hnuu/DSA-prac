```cpp
/*
class Node
{
  public:
    int data;
    Node * next;
    Node (int x)
    {
        data=x;
        next=NULL;
    }

};
*/

class Solution {
  public:
    bool isSorted(Node* head) {
        if(!head || !head->next) return true;

        while(head && head->next->data==head->data) head=head->next;

        if(!head->next) return true;

        Node *prev=nullptr;
        bool isInc=head->next->data>head->data;
        while(head){
            if(prev && (isInc ? prev->data>head->data:prev->data<head->data)) return false;
            prev=head;
            head=head->next;
        }
        return true;
    }
};
```
