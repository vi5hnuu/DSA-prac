class Solution {
  public:
    int getKthFromLast(Node* head, int k) {
        if(!head) return -1;
        int len=lenLL(head);
        if(k>len) return -1;
        int st=len-k;
        while(st>0) {
            head=head->next;
            st--;
        }
        return head->data;
    }
  private:
    int lenLL(Node* node){
        int len=0;
        while(node){
            node=node->next;
            len++;
        }
        return len;
    }
};

class Solution {
  public:
    int getKthFromLast(Node* head, int k) {
        if(!head) return -1;

        Node *first=head,*second=head;

        while(second && k>0){
            second=second->next;
            k--;
        }
        if(k!=0) return -1;

        while(second){
            first=first->next;
            second=second->next;
        }
        return first->data;
    }
};
