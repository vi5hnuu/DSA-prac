```cpp
/*
class Node {
  public:
    int data;
    Node* next;

    // Default constructor
    Node() {
        data = 0;
        next = NULL;
    }

    // Parameterised Constructor
    Node(int data) {
        this->data = data;
        this->next = NULL;
    }
};
*/
class Solution {
  public:
    vector<int> printList(Node *head) {
        vector<int> nodes;
        while(head){
            nodes.push_back(head->data);
            head=head->next;
        }
        return nodes;
    }
};
```

```cpp
/*
class Node {
  public:
    int data;
    Node* next;

    // Default constructor
    Node() {
        data = 0;
        next = NULL;
    }

    // Parameterised Constructor
    Node(int data) {
        this->data = data;
        this->next = NULL;
    }
};
*/
class Solution {
  public:
    vector<int> printList(Node *head) {
        vector<int> nodes;
        fillLL(nodes,head);
        return nodes;
    }
  private:
    void fillLL(vector<int> &nodes,const Node *const head){
        if(!head) return;
        nodes.push_back(head->data);
        fillLL(nodes,head->next);
    }
};
```
