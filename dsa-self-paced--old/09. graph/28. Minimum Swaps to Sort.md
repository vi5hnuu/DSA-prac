class Solution {
  public:
    // Function to find the minimum number of swaps required to sort the array.
    int minSwaps(vector<int>& arr) {
        int sz=arr.size();
        vector<pair<int,int>> elementIdx;
        for(int i=0;i<sz;i++) elementIdx.push_back(make_pair(arr[i],i));
        sort(elementIdx.begin(),elementIdx.end());

        int minSwaps=0;
        vector<bool> visited(sz,false);

        for(int i=0;i<sz;i++){
            auto pos=elementIdx[i].second;
            if(visited[pos]) continue;

            int cycle=0;
            while(!visited[pos]){
                visited[pos]=true;
                pos=elementIdx[pos].second;
                cycle++;
            }
            minSwaps+=(cycle/2 ? cycle-1 : 0);
        }
        return minSwaps;
    }
};
