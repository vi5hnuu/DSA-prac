```cpp
class Solution {
  public:
    bool isCyclic(int V, vector<vector<int>> &edges) {
        vector<vector<int>> adj(V,vector<int>());
        for(auto &edge:edges){
            adj[edge[0]].push_back(edge[1]);
        }
        vector<bool> visited(V,false);
        vector<bool> backedge(V,false);
        for(int i=0;i<V;i++){
            if(visited[i]) continue;
            if(isCyle(adj,visited,backedge,i)) return true;
        }
        return false;
    }
  private:
    bool isCyle(vector<vector<int>> &adj,vector<bool> &visited,vector<bool> &backedge,int root){
        if(visited[root]) return false;
        backedge[root]=visited[root]=true;
        for(int neighbour:adj[root]){
            if(visited[neighbour]){
                if(backedge[neighbour]) return true;
                continue;
            }
            if(isCyle(adj,visited,backedge,neighbour)) return true;
        }
        backedge[root]=false;
        return false;
    }
};
```

### topological sorting (kahn's algorithm)

```cpp
class Solution {
  public:
    bool isCyclic(int V, vector<vector<int>> &edges) {
        vector<vector<int>> adj(V,vector<int>());
        for(auto &edge:edges){
            adj[edge[0]].push_back(edge[1]);
        }

        //topological sort
        vector<int> indegree(V,0);
        for(auto &edge:edges){
            indegree[edge[1]]++;
        }
        queue<int> nodes;
        for(int i=0;i<V;i++){
            if(!indegree[i]) nodes.push(i);
        }
        while(!nodes.empty()){
            int node=nodes.front();
            nodes.pop();

            for(int neighbour:adj[node]){
                indegree[neighbour]--;
                if(!indegree[neighbour]) nodes.push(neighbour);
            }
        }
        for(int i=0;i<V;i++){
            if(indegree[i]) return true;
        }
        return false;
    }
};
```

```cpp
class Solution {
  public:
    bool isCyclic(int V, vector<vector<int>> &edges) {
        vector<vector<int>> adj(V,vector<int>());
        for(auto &edge:edges){
            adj[edge[0]].push_back(edge[1]);
        }

        //topological sort
        vector<int> indegree(V,0);
        for(auto &edge:edges){
            indegree[edge[1]]++;
        }
        queue<int> nodes;
        for(int i=0;i<V;i++){
            if(!indegree[i]) nodes.push(i);
        }
        int indegreeNodesCount=nodes.size();
        while(!nodes.empty()){
            int node=nodes.front();
            nodes.pop();

            for(int neighbour:adj[node]){
                indegree[neighbour]--;
                if(!indegree[neighbour]) {
                    indegreeNodesCount++;
                    nodes.push(neighbour);
                }
            }
        }
        return indegreeNodesCount!=V;
    }
};
```
