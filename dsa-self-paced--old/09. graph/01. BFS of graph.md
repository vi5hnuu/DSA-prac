class Solution {
  public:
    vector<int> bfs(vector<vector<int>> &adj) {
        if(!adj.size()) return {};
        vector<int> bfs;

        vector<bool> visited(adj.size(),false);
        queue<int> nodes;
        nodes.push(0);
        visited[0]=true;

        while(!nodes.empty()){
            int sz=nodes.size();
            while(sz--){
                int node=nodes.front();
                bfs.push_back(node);
                nodes.pop();
                for(int &neighbour:adj[node]){
                    if(visited[neighbour]) continue;
                    visited[neighbour]=true;
                    nodes.push(neighbour);
                }
            }
        }
        return bfs;
    }
};
