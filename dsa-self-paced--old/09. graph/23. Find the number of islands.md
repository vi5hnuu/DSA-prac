```cpp
class Solution {
  public:
    int countIslands(vector<vector<char>>& grid) {
        int islandCount=0;
        int r=grid.size();
        int c=grid[0].size();

        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(grid[i][j]!='L') continue;
                floodLand(grid,r,c,i,j);
                islandCount++;
            }
        }
        return islandCount;
    }
  private:
    void floodLand(vector<vector<char>> &grid,int r,int c,int i,int j){
        if(i<0 || j<0 || i>=r || j>=c || grid[i][j]!='L') return;
        grid[i][j]='W';

        int dx[]={1,1,-1,-1,1,-1,0,0};
        int dy[]={-1,1,-1,1,0,0,-1,1};
        for(int k=0;k<8;k++){
            int x=i+dx[k];
            int y=j+dy[k];
            if(x>=0 && y>=0 && x<r && y<c && grid[x][y]=='L') floodLand(grid,r,c,x,y);
        }
    }
};
```

```cpp
class Solution {
  public:
    int countIslands(vector<vector<char>>& grid) {
        int islandCount=0;
        int r=grid.size();
        int c=grid[0].size();

        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(grid[i][j]!='L') continue;
                floodLand(grid,r,c,i,j);
                islandCount++;
            }
        }
        return islandCount;
    }
  private:
    void floodLand(vector<vector<char>> &grid,int r,int c,int i,int j){
        if(i<0 || j<0 || i>=r || j>=c || grid[i][j]!='L') return;

        queue<pair<int,int>> nodes;
        grid[i][j]='W';
        nodes.push(make_pair(i,j));

        while(!nodes.empty()){
            int sz=nodes.size();
            while(sz--){
                auto [i,j]=nodes.front();
                nodes.pop();
                int dx[]={1,1,-1,-1,1,-1,0,0};
                int dy[]={-1,1,-1,1,0,0,-1,1};
                for(int k=0;k<8;k++){
                    int x=i+dx[k];
                    int y=j+dy[k];
                    if(x>=0 && y>=0 && x<r && y<c && grid[x][y]=='L') {
                        grid[x][y]='W';
                        nodes.push(make_pair(x,y));
                    }
                }
            }
        }
    }
};
```
