```cpp
class Solution {
  public:
    // Function to count paths between two vertices in a directed graph.
    int countPaths(int V, vector<int> adj[], int source, int destination) {
        if(source==destination) return 1;

        int paths=0;
        for(auto neighbour:adj[source]){
            paths+=countPaths(V,adj,neighbour,destination);
        }
        return paths;
    }
};
```

```cpp
class Solution {
  public:
    int countPaths(int V, vector<int> adj[], int source, int destination) {
       queue<vector<int>> q,res;
       q.push({source});

       while(q.empty() == false) {
           vector<int> a = q.front(); q.pop();
           int n = a.size();
           if(a[n-1] == destination) {
               res.push(a);
               continue;
           }
           for(auto x:adj[a[n-1]]) {
                   a.push_back(x);
                   q.push(a);
                   a.pop_back();
           }
       }
       return res.size();
    }
};
```
