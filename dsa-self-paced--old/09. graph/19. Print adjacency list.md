class Solution {
  public:
    // Function to return the adjacency list for each vertex.
    vector<vector<int>> printGraph(int V, vector<pair<int, int>>& edges) {
        vector<vector<int>> adjL(V,vector<int>());

        for(auto &edge:edges){
            int u=edge.first;
            int v=edge.second;

            adjL[u].push_back(v);
            adjL[v].push_back(u);
        }
        return adjL;
    }
};
