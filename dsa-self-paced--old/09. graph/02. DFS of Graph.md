```cpp
class Solution {
  public:
    vector<int> dfs(vector<vector<int>>& adj) {
        vector<int> dfs;
        vector<bool> visited(adj.size(),false);

        fillDfs(adj,visited,dfs,0);
        return dfs;
    }
  private:
    void fillDfs(vector<vector<int>> &adj,vector<bool> &visited,vector<int> &dfs,int root){
        if(visited[root] || root>=adj.size()) return;
        visited[root]=true;
        dfs.push_back(root);

        for(auto neighbour:adj[root]){
            if(visited[neighbour]) continue;
            fillDfs(adj,visited,dfs,neighbour);
        }
    }
};
```

```cpp
class Solution {
  public:
    vector<int> dfs(vector<vector<int>>& adj) {
        vector<int> dfs;
        vector<bool> visited(adj.size(),false);

        stack<int> stck;
        stck.push(0);

        while(!stck.empty()){
            int tp=stck.top();
            stck.pop();
            if(visited[tp]) continue;
            dfs.push_back(tp);
            visited[tp]=true;
            for(int i=adj[tp].size()-1;i>=0;i--){
                int neighbour=adj[tp][i];
                stck.push(neighbour);
            }
        }
        return dfs;
    }
};
```
