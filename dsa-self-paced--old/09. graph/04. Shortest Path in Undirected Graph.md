class Solution {
  public:
    // Function to find the shortest path from source to all other nodes
    vector<int> shortestPath(vector<vector<int>>& adj, int src) {
        vector<int> shortestPaths(adj.size(),-1);
        vector<bool> visited(adj.size(),false);
        visited[src]=true;
        shortestPaths[src]=0;
        queue<int> nodes;
        nodes.push(src);

        int path=-1;
        while(!nodes.empty()){
            int sz=nodes.size();
            while(sz--){
                int node=nodes.front();
                nodes.pop();
                shortestPaths[node]=path+1;

                for(int neighbour:adj[node]){
                    if(visited[neighbour]) continue;
                    visited[neighbour]=true;
                    nodes.push(neighbour);
                }
            }
            path++;
        }

        return shortestPaths;
    }
};

class Solution {
  public:
    // Function to find the shortest path from source to all other nodes
    vector<int> shortestPath(vector<vector<int>>& adj, int src) {
        vector<int> shortestPaths(adj.size(),-1);
        vector<bool> visited(adj.size(),false);
        shortestPaths[src]=0;
        queue<int> nodes;
        nodes.push(src);

        while(!nodes.empty()){
            int sz=nodes.size();
            while(sz--){
                int node=nodes.front();
                nodes.pop();
                for(int neighbour:adj[node]){
                    if(shortestPaths[neighbour]>=0) continue;
                    nodes.push(neighbour);
                    shortestPaths[neighbour]=shortestPaths[node]+1;
                }
            }
        }

        return shortestPaths;
    }
};
