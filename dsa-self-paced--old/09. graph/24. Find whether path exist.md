class Solution {
  public:
    // Function to find whether a path exists from the source to destination.
    bool is_Possible(vector<vector<int>>& grid) {
        int r=grid.size();
        int c=grid[0].size();

        queue<pair<int,int>> nodes;

        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                if(grid[i][j]==1){
                    nodes.push({i,j});
                    break;
                }
            }
        }

        while(!nodes.empty()){//move in circle
            int sz=nodes.size();

            while(sz--){
                auto [i,j]=nodes.front();
                nodes.pop();

                int dx[]={1,-1,0,0};
                int dy[]={0,0,-1,1};
                for(int k=0;k<4;k++){
                    int x=i+dx[k];
                    int y=j+dy[k];
                    if(x>=0 && y>=0 && x<r && y<c){
                        if(grid[x][y]==2) return true;
                        else if(grid[x][y]==3){
                            grid[x][y]=0;//so that it won't be pushed again
                            nodes.push({x,y});
                        }
                    }
                }
            }
        }

        return false;
    }
};
