class Solution {
  public:
    // Function to count the number of digits in n that evenly divide n
    int evenlyDivides(int n) {
        int valN=n;
        int count=0;
        while(n){
            int digit=n%10;
            n/=10;
            if(digit && valN%digit==0) count++;
        }
        return count;
    }
};
