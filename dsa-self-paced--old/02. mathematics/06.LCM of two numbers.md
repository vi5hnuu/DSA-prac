### Lowest number which can be divided by both a and b

```cpp
class Solution {
  public:
    int lcm(int a, int b) {

      int lcm=max(a,b);
      while(true){
          if(lcm%a==0 && lcm%b==0) return lcm;
          lcm++;
      }
      return -1;
    }
};
```

```cpp
class Solution {
  public:
    int lcm(int a, int b) {
        // axb=gcd(a,b)xLcm
      return a*b/gcd(a,b);
    }
    int gcd(int a, int b) {
        if(!a || !b) return max(a,b);
        return gcd(b%a,a);
    }
};
```
