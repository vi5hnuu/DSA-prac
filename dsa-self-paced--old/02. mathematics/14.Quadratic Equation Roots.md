```cpp
class Solution {
  public:
    vector<int> quadraticRoots(int a, int b, int c) {
        const double q=b*b-4*a*c;
        if(q<0) return {-1};
        const double qq=sqrt(q);
        return {floor((-b+qq)/(2*a)),floor((-b-qq)/(2*a))};
    }
};
```
