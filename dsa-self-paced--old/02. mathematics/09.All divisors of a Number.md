### TLE

```cpp
class Solution {
  public:
    void print_divisors(int n) {
       for(int i=1;i<=n;i++){
           if(n%i==0) cout<<i<<" ";
       }
    }
};
```

### OR

```cpp
class Solution {
  public:
    void print_divisors(int n) {
       int i=1;
       for(;i*i<n;i++){
           if(n%i==0) cout<<i<<" ";
       }
       if(i*i!=n) i--;
       for(;i>=1;i--){
           if(n%i==0) cout<<n/i<<" ";
       }
    }
};
```
