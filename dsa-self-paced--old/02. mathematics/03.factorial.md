```cpp
class Solution {
  public:
    int factorial(int n) {
        if(n<=0) return 0;
        int fact=n;
        for(int i=n-1;i>=1;i--) fact=(fact*i)%INT_MAX;
        return fact;
    }
};
```

### OR

```cpp
class Solution {
  public:
    int factorial(int n) {
        if(n<=0) return 0;
        if(n==1) return 1;
        return n*factorial(n-1);
    }
};
```
