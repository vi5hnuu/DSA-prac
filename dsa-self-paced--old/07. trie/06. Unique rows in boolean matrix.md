class TrieNode{
  public:
    bool isWord=false;
    TrieNode* childs[2]={nullptr};

    TrieNode(){}
};

class Trie {
  TrieNode *root;
  public:

    Trie() {
        root=new TrieNode();
    }

    ~Trie(){
        burnTrie(root);
    }

    bool insert(int binArr[],int sz) {
        int i=0;
        TrieNode* rt=root;
        while(i<sz){
            if(((rt->childs)[binArr[i]])!=NULL){
                rt=(rt->childs)[binArr[i]];
            }else{
                rt=(rt->childs)[binArr[i]]=new TrieNode();
            }
            i++;
        }
        if(rt->isWord) return false;//already there
        rt->isWord=true;
        return true;//inserted
    }
  private:
    void burnTrie(TrieNode *root){
        if(!root) return;
        for(int i=0;i<2;i++){
            if(root->childs[i]) burnTrie(root->childs[i]);
        }
        delete root;
    }
};

class Solution {
  public:
    // #define MAX 1000
    vector<vector<int>> uniqueRow(int M[MAX][MAX], int row, int col) {
        Trie trie;
        vector<vector<int>> uniqueRows;

        for(int i=0;i<row;i++){
            if(trie.insert(M[i],col)){
                uniqueRows.push_back(vector<int>(M[i],M[i]+col));
            }
        }

        return uniqueRows;
    }
};
