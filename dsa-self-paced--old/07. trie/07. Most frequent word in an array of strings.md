class TrieNode {
public:
    TrieNode* child[26];
    bool isWord;
    int wordFreq;
    int firstIndex;
    string word; // store full word at terminal

    TrieNode() {
        isWord = false;
        wordFreq = 0;
        firstIndex = -1;
        for(int i = 0; i < 26; i++) child[i] = nullptr;
    }
};

class Trie {
    TrieNode* root;

    // global best result
    string frequentWord = "";
    int bestFreq = 0;
    int bestPos = -1;  // first occurrence position
public:
    Trie() {
        root = new TrieNode();
    }

    string freqWord(){
        return frequentWord;
    }

    void insertAndUpdateFrequentWord(string &s, int index) {
        TrieNode* curr = root;
        for(char c : s) {
            int id = c - 'a';
            if(!curr->child[id]) curr->child[id] = new TrieNode();
            curr = curr->child[id];
        }

        // now at terminal
        if(!curr->isWord) {
            curr->isWord = true;
            curr->word = s;
            curr->firstIndex = index;
        }

        curr->wordFreq++;
        updateFrequentWord(curr);
    }
  private:
    void updateFrequentWord(TrieNode* node) {
        int f = node->wordFreq;
        int pos = node->firstIndex;

        if(f > bestFreq || (f == bestFreq && pos > bestPos)) {
            bestFreq = f;
            bestPos = pos;
            frequentWord = node->word;
        }
    }
};

class Solution {
public:
    string mostFrequentWord(string arr[], int n) {
        Trie trie;

        for(int i = 0; i < n; i++)
            trie.insertAndUpdateFrequentWord(arr[i], i);

        return trie.freqWord();
    }
};
