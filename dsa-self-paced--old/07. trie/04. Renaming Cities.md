// User function template for C++

class TrieNode{
  public:
    bool isWord;
    int freq;
    vector<TrieNode*> *childs;

    TrieNode(){
        this->isWord=false;
        this->freq=1;
        childs=new vector<TrieNode*>(26,nullptr);
    }

    ~TrieNode(){
        delete childs;
    }
};
class Trie {
  TrieNode *root;
  public:

    Trie() {
        root=new TrieNode();
    }

    ~Trie(){
        burnTrie(root);
    }

    string insertAndName(string &word) {
        int sz=word.length();
        int i=0;
        TrieNode* rt=root;
        string name="";

        bool unique=false;
        int minFreq=INT_MAX;
        while(i<sz){
            if(!unique) name+=word[i];
            if(((*rt->childs)[word[i]-'a'])!=NULL){
                rt=(*rt->childs)[word[i]-'a'];
                minFreq=min(minFreq,rt->freq);
                rt->freq++;
            }else{
                unique=true;
                rt=(*rt->childs)[word[i]-'a']=new TrieNode();
            }
            i++;
        }
        if(rt->isWord) return name+' '+to_string(minFreq+1);
        rt->isWord=true;
        return name;
    }
  private:
    void burnTrie(TrieNode *root){
        if(!root) return;
        for(TrieNode* child:*root->childs){
            if(child) burnTrie(child);
        }
        delete root;
    }
};

class Solution {
  public:
    void check(string *arr, int n) {
       Trie trie;
       for(int i=0;i<n;i++){
           cout<<trie.insertAndName(arr[i])<<endl;
       }
    }
};
