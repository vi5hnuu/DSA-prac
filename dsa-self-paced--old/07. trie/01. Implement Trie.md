// User function template for C++
class TrieNode{
  public:
    bool isWord;
    vector<TrieNode*> *childs;

    TrieNode(){
        this->isWord=false;
        childs=new vector<TrieNode*>(26,nullptr);
    }

    ~TrieNode(){
        delete childs;
    }
};
class Trie {
  TrieNode *root;
  public:

    Trie() {
        root=new TrieNode();
    }

    ~Trie(){
        burnTrie(root);
    }

    void insert(string &word) {
        insert(word,0);
    }

    bool search(string &word) {
        return search(word,0);
    }

    bool isPrefix(string &word) {
        return search(word,0,true);
    }
  private:
    void insert(string &word,int i){
        int sz=word.length();
        if(i>=sz) return;
        TrieNode* rt=root;
        while(i<sz){
            if(((*rt->childs)[word[i]-'a'])!=NULL){
                rt=(*rt->childs)[word[i]-'a'];
            }else{
                rt=(*rt->childs)[word[i]-'a']=new TrieNode();
            }
            i++;
        }
        rt->isWord=true;
    }
    bool search(string &word,int i,bool prefix=false){
        int sz=word.length();
        if(i>=sz) return false;
        TrieNode* rt=root;
        while(i<sz && ((*rt->childs)[word[i]-'a'])!=NULL){
            rt=(*rt->childs)[word[i]-'a'];
            i++;
        }
        return i>=sz && (prefix || rt->isWord);
    }
    void burnTrie(TrieNode *root){
        if(!root) return;
        for(TrieNode* child:*root->childs){
            if(child) burnTrie(child);
        }
        delete root;
    }
};
