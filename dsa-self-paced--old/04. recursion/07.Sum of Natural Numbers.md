```cpp
class Solution {
  public:
    int findSum(int n) {
        int sm=0;
        for(int i=1;i<=n;i++) sm+=i;
        return sm;
    }
};
```

```cpp
class Solution {
  public:
    int findSum(int n) {
        if(n<=0) return 0;
        return n+findSum(n-1);
    }
};
```

```cpp
class Solution {
  public:
    int findSum(int n,int acc=0) {
        if(n<=0) return acc;
        return findSum(n-1,acc+n);
    }
};
```

```cpp
class Solution {
  public:
    int findSum(int n) {
        if(n<=0) return 0;
        return n*(n+1)/2;
    }
};
```
