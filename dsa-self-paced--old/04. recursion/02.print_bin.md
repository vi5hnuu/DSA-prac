## print binary representation of x

```cpp
 #include<bits/stdc++.h>
 using namespace std;


void printBin(int);

 int main(){
    int n; cin>>n;
    printBin(n);
  }


  void printBin(int n){
    if(!n){
      cout<<"0";
    }

    string s="";
    while(n){
      s= to_string(n%2)+s;
      n/=2;
    }
    cout<<s;
  }
```
