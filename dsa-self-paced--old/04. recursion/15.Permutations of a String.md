```cpp
class Solution {
  public:
    vector<string> findPermutation(string &s) {
        int sz=s.length();
        if(!sz) return {};
        set<string> permutations;//set to remove duplicates
        vector<bool> usedMp(sz,false);//map to avoid using used items again

        string permuteTmp="";
        permuteTmp.reserve(s.length());

        fillPermutes(s,usedMp,0,permutations,permuteTmp);
        return vector<string>(permutations.begin(),permutations.end());
    }
    void fillPermutes(const string &s,vector<bool> &usedMp,int idx,set<string> &permutations,string &permute){
        int slen=s.length();
        if(idx>=slen) {
            permutations.insert(permute);
            return;
        }
        for(int i=0;i<slen;i++){
            if(usedMp[i]) continue;


            usedMp[i]=true;
            permute += s[i];

            fillPermutes(s,usedMp,idx+1,permutations,permute);

            permute.pop_back();
            usedMp[i]=false;
        }
    }
};
```

```cpp
class Solution {
  public:
    vector<string> findPermutation(string &s) {
        int sz=s.length();
        if(!sz) return {};
        // Sort to group duplicates together
        sort(s.begin(), s.end());

        vector<string> permutations;
        vector<bool> usedMp(sz,false);

        string permuteTmp="";
        permuteTmp.reserve(s.length());

        fillPermutes(s,usedMp,permutations,permuteTmp);
        return vector<string>(permutations.begin(),permutations.end());
    }
    void fillPermutes(const string &s,vector<bool> &usedMp,vector<string> &permutations,string &permute){
         int slen=s.length();
         // Base case
        if (permute.length() == slen) {
            permutations.push_back(permute);
            return;
        }

        for(int i=0;i<slen;i++){
            if(usedMp[i]) continue;

            // OPTIMIZATION: Skip duplicates at the same recursion level [allow only forward use]
            if (i > 0 && s[i] == s[i-1] && !usedMp[i-1]) {
                continue;
            }

            usedMp[i]=true;
            permute += s[i];

            fillPermutes(s,usedMp,permutations,permute);

            permute.pop_back();
            usedMp[i]=false;
        }
    }
};
```
