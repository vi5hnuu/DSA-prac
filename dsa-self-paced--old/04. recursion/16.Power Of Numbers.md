```cpp
class Solution {
  public:
    int reverseExponentiation(int n) {
        int rev=reverseNum(n);
        return (int)power(n,rev);
    }
    double power(int x,int y){
        if(y<0) return power(1.0/x,y);
        if(!y) return 1;

        int p=power(x,y/2);
        return p*p*(y&1 ? x:1);
    }
    int reverseNum(int n){
        int rev=0;
        while(n){
            rev=rev*10+n%10;
            n/=10;
        }
        return rev;
    }
};
```

```cpp
class Solution {
  public:
    int reverseExponentiation(int n) {
        int rev=reverseNum(n);
        return (int)power(n,rev);
    }
    int power(int x,int y){
        if(y<0) return power(1.0/x,y);
        if(!y) return 1;

        int mul=x;
        int res=1;
        while(y){
            if(y&1) res=(res*mul)%INT_MAX;
            y>>=1;
            mul=mul*mul;
        }
        return res;
    }
    int reverseNum(int n){
        int rev=0;
        while(n){
            rev=rev*10+n%10;
            n/=10;
        }
        return rev;
    }
};
```
