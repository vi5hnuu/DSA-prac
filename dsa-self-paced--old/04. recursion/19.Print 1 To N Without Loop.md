```cpp
class Solution {
  public:
    void printNos(int n) {
        if(n<=0) return;
        printNos(n-1);
        cout<<n<<" ";
    }
};
```

### TAIL RECURSION

```cpp
class Solution {
  public:
    void printNos(int n,int acc=1) {
        if(n<=0) return;
        cout<<acc<<" ";
        printNos(n-1,acc+1);
    }
};
```
