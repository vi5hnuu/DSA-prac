```cpp
class Solution {
  public:
    int sumOfDigits(int n) {
        int sm=0;
        while(n){
            sm+=n%10;
            n/=10;;
        }
        return sm;
    }
};
```

```cpp
class Solution {
  public:
    int sumOfDigits(int n) {
        if(n<=0) return 0;
        return (n%10) + sumOfDigits(n/10);
    }
};
```

```cpp
class Solution {
  public:
    int sumOfDigits(int n,int acc=0) {
        if(n<=0) return acc;
        return sumOfDigits(n/10,acc+(n%10));
    }
```
