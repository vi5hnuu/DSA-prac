https://www.geeksforgeeks.org/problems/factorial5739/1

```cpp
class Solution {
  public:
    int factorial(int n) {
        if(n<=0) return 0;
        if(n==1) return 1;
        return n*factorial(n-1);
    }
};
```

### Tail recursion

```cpp
class Solution {
  public:
    int factorial(int n,int acc=1) {
        if(n<=1) return acc;
        return factorial(n-1,acc*n);
    }
};
```
