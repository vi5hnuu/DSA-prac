```cpp
class Solution {
  public:
    bool isLucky(int n) {
      vector<int> nums;
      for(int i=1;i<=n;i++) nums.push_back(i);

      int shift=2;
      while(nums.size()>=shift){
          vector<int> remains;

          for(int i=0,nsz=nums.size();i<nsz;i++){
              if((i+1)%shift==0) {
                  if(nums[i]==n) return false;
                  continue;
              }
              remains.push_back(nums[i]);
          }
          nums=remains;
          shift++;
      }
      return true;
    }
};
```

```cpp
class Solution {
  public:
    bool isLucky(int n) {
      for(int i=2;i<=n;i++){
          if(n%i==0) return false;
          n=n-n/i;
      }
      return true;
    }
};
```

```cpp
class Solution {
  public:
    bool isLucky(int n) {
      return isLucky(n,2);
    }
  private:
    bool isLucky(int n,int step) {
        if(n%step==0) return false;
        if(n<step) return true;
        return isLucky(n-n/step,step+1);
    }
};
```
