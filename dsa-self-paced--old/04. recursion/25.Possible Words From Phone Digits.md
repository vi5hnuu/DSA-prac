class Solution {
  public:
    vector<string> possibleWords(vector<int> &arr) {
       vector<string> words;

       map<int,string> keys={{0,""},{1,""},{2,"abc"},{3,"def"},{4,"ghi"},{5,"jkl"},{6,"mno"},{7,"pqrs"},{8,"tuv"},{9,"wxyz"}};
       string wordTmp;
       fillWords(keys,words,arr,wordTmp);
       return words;
    }
  private:
    void fillWords(const map<int,string> &keys,vector<string> &words,vector<int> &arr,string &wordTmp,int idx=0){
        if(idx>=arr.size()){
            words.push_back(wordTmp);
            return;
        }

        const string key=keys.at(arr[idx]);
        if(!key.length()){
            fillWords(keys,words,arr,wordTmp,idx+1);
        }
        for(int i=0;i<key.length();i++){
            wordTmp.push_back(key[i]);
            fillWords(keys,words,arr,wordTmp,idx+1);
            wordTmp.pop_back();
        }
    }
};
