```cpp
class Solution {
  public:
    int digitalRoot(int n) {
        int sm=0;
        while(n){
            sm+=n%10;
            n/=10;
            if(!n && sm>9){
                n=sm;
                sm=0;
            }
        }
        return sm;
    }
};
```
