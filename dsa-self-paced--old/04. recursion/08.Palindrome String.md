```cpp
class Solution {
  public:
    bool isPalindrome(string& s) {
        int len=s.length();
        for(int i=0;i<len/2;i++){
            if(s[i]!=s[len-1-i]) return false;
        }
        return true;
    }
};
```

```cpp
class Solution {
  public:
    bool isPalindrome(string& s,int idx=0) {
        int len=s.length();
        if(idx>=len/2) return true;
        if(s[idx]!=s[len-1-idx]) return false;
        return isPalindrome(s,idx+1);
    }
};
```
