```cpp
class Solution {
  public:
    int RecursivePower(int n,int p) {
        return (int)power(n,p);
    }
    double power(int x,int y){
        if(y<0) return power(1.0/x,y);
        if(!y) return 1;

        int p=power(x,y/2);
        return p*p*(y&1 ? x:1);
    }
};
```

```cpp
#include<bits/stdc++.h>
using namespace std;

class Solution {
  public:
    int RecursivePower(int n,int p) {
        return (int)power(n,p);
    }
    int power(int x,int y){
        if(y<0) return power(1.0/x,y);
        if(!y) return 1;

        int mul=x;
        int res=1;
        while(y){
            if(y&1) res=(res*mul)%INT_MAX;
            y>>=1;
            mul=mul*mul;
        }
        return res;
    }
};
```
