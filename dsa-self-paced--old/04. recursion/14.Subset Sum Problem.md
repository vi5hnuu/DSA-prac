class Solution {
  public:
    bool isSubsetSum(vector<int>& arr, int sum) {
        int sz=arr.size();
        // int totalSubsets=1<<sz;//1 <= arr.size() <= 200
        long totalSubsets=pow(2,sz);//since arr.size can be larger it fails for large n as long cannot store large value

        for(long i=1;i<totalSubsets;i++){
            int subset=sum;
            for(int j=0;j<sz;j++){
                if((i>>j)&1) subset-=arr[j];
                if(!(i>>j)) break;//no need to check further bits
                if(subset<0) break;
            }
            if(!subset) return true;
        }
        return false;
    }
};

class Solution {
  public:
    bool isSubsetSum(vector<int>& arr, int sum) {
        return isSubsetSum(arr,0,sum);
    }
  private:
    bool isSubsetSum(vector<int>& arr,int idx, int sum) {
        if(!sum) return true;
        if(sum<0 || idx>=arr.size()) return false;
        return isSubsetSum(arr,idx+1,sum-arr[idx]) || isSubsetSum(arr,idx+1,sum);
    }
};

        unordered_map<string,int> memo;
        return isSubsetSum(memo,arr,0,sum);
    }
  private:
    bool isSubsetSum(unordered_map<string,int> &memo,vector<int>& arr,int idx, int sum) {
        if(!sum) return true;
        if(sum<0 || idx>=arr.size()) return false;

        string key = to_string(idx) + "," + to_string(sum);
        if (memo.find(key) != memo.end()) {
            return memo[key];
        }
        return (memo[key] = (isSubsetSum(memo,arr,idx+1,sum-arr[idx]) || isSubsetSum(memo,arr,idx+1,sum)));
    }
};

class Solution {
  public:
    bool isSubsetSum(vector<int>& arr, int sum) {
        //H : 0..sum
        //V : arr[i]

        int sz=arr.size();

        //row is array elements
        //cols are 0...sum
        vector<vector<bool>> dp(sz,vector<bool>(sum+1,false));

        //yes we can create 0 sum by not using any el
        for (int r = 0; r < sz; r++) dp[r][0] = true;

        //yes we can create sum 0....n if arr[0]==c
        for (int c = 0; c <= sum; c++) dp[0][c] = c==arr[0];

        //i can make sum c if sum c is already made my using all elements r-1>=0 OR sum c- current array element is already made using r-1>=0 arrray elements
        for(int c=1;c<=sum;c++){
            for(int r=1;r<sz;r++){
                dp[r][c]=dp[r-1][c] || (c-arr[r]>=0 && dp[r-1][c-arr[r]]);
            }
        }
        return dp[sz-1][sum];
    }
};

class Solution {
  public:
    bool isSubsetSum(vector<int>& arr, int sum) {
        //H : 0..sum
        //V : arr[i]

        int sz=arr.size();
        vector<bool> dp(sum+1,false);
        dp[0]=true;

        for(int i=0;i<sz;i++){
            for(int j=sum;j>=arr[i];j--){
                dp[j]=dp[j] || dp[j-arr[i]];
                if(dp[j] && j==sum) return true;
            }
        }
        return dp[sum];
    }
};
