class Solution {
  public:
    int maxConsecutiveOnes(int n) {
       int mx=0;

       int cnt=0;
       while(n){
           if(n&1) cnt++;
           else{
               mx=max(mx,cnt);
               cnt=0;
           }
           n>>=1;
       }
       mx=max(mx,cnt);
       return mx;
    }
};
