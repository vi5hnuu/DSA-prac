```cpp
class Solution {
  public:
    int getOddOccurrence(vector<int>& arr) {
       int sz=arr.size();
       for(int n:arr){
           int cnt=0;
           for(int x:arr){
               if(x==n) cnt++;
           }
           if(cnt&1) return n;
       }
       return -1;
    }
};
```

```cpp
class Solution {
  public:
    int getOddOccurrence(vector<int>& arr) {
        map<int,int> freq;

        for(int &n:arr) freq[n]++;

        for(auto &f:freq){
            if(f.second&1) return f.first;
        }
        return -1;
    }
};
```

```cpp
class Solution {
  public:
    int getOddOccurrence(vector<int>& arr) {
        int allXor=0;
        for(int &n:arr) allXor^=n;
        return allXor;
    }
};
```
