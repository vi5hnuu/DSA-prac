```cpp
class Solution {
  public:
    int getFirstSetBit(int n) {
        if(!n) return 0;

        int pos=1;
        while(n && (n&1)==0) {
            pos++;
            n>>=1;
        }
        return pos;
    }
};
```

```cpp
class Solution {
  public:
    int getFirstSetBit(int n) {
        n=n & ~(n-1);//turn off all bits except first
        return log2(n)+1;
    }
};
```
