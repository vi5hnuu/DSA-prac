```cpp
class Solution {
  public:
    bool isSparse(int n) {
        int conseCnt=0;

        while(n){
            if(n&1) conseCnt++;
            else conseCnt=0;
            if(conseCnt>=2) return false;
            n>>=1;
        }
        return true;
    }
};
```

class Solution {
  public:
    bool isSparse(int n) {
        return (n & (n >> 1)) == 0;
    }
};
