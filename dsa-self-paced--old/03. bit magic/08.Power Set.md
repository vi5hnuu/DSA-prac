```cpp
class Solution {
  public:
    vector<string> AllPossibleStrings(string s) {
        int sz=s.length();
        if(!sz) return {};
        vector<string> subsets;
        fillSubsets(s,0,subsets);
        sort(subsets.begin(),subsets.end());
        return subsets;
    }
    void fillSubsets(string s,int from,vector<string> &subsets,string subset=""){
        if(from>=s.length()) {
            subsets.push_back(subset);
            return;
        }
        fillSubsets(s,from+1,subsets,subset);
        fillSubsets(s,from+1,subsets,subset+s[from]);
    }
};
```

```cpp
class Solution {
  public:
    vector<string> AllPossibleStrings(string s) {
        int sz=s.length();
        if(!sz) return {};
        const int totalSubsets=1<<sz;

        vector<string> subsets;
        for(int i=1;i<=totalSubsets;i++){
            string subset="";
            for(int j=0;j<sz;j++){
                if((i>>j)&1) subset+=s[j];
            }
            subsets.push_back(subset);
        }
        sort(subsets.begin(),subsets.end());
        return subsets;
    }
};
```
