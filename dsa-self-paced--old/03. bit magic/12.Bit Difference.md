```cpp
class Solution {
  public:
    int countBitsFlip(int a, int b) {
        int bitDiff=0;
        for(int i=0;i<=31;i++){
            if((a&(1<<i))!=(b&(1<<i))) bitDiff++;
        }
        return bitDiff;
    }
};
```

```cpp
class Solution {
  public:
    int countBitsFlip(int a, int b) {
        int lastBitPos=log2(max(a,b));//0 based
        int bitDiff=0;
        for(int i=0;i<=lastBitPos;i++){
            if((a&(1<<i))!=(b&(1<<i))) bitDiff++;
        }
        return bitDiff;
    }
};
```
