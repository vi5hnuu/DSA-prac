```cpp
class Solution {
  public:
    int posOfRightMostDiffBit(int m, int n) {
        int val=m^n;
        if(!val) return -1;
        int turnOffAllExceptFirstLSB=val& ~(val-1);
        return log2(turnOffAllExceptFirstLSB)+1;
    }
};
```
