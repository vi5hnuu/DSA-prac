```cpp
class Solution {
  public:
    bool isPowerofTwo(int n) {
        if(n<=0) return false;
        while(n && n%2==0) n>>=1;
        return n==1;
    }
};
```

```cpp
class Solution {
  public:
    bool isPowerofTwo(int n) {
        return n>0 && (n&(n-1))==0;
    }
};
```
