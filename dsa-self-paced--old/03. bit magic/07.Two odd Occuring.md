```cpp
class Solution {
  public:
    vector<int> twoOddNum(vector<int>& arr) {
        set<int> res;
        for(int n:arr){
            int cnt=0;
            for(int x:arr) cnt+=(x==n);
            if(cnt&1) res.insert(n);
        }

        vector<int> r(res.begin(),res.end());
        sort(r.begin(),r.end(),std::greater<int>());
        return r;
    }
};
```

```cpp
class Solution {
  public:
    vector<int> twoOddNum(vector<int>& arr) {
        int allXor=0;
        for(int n:arr) allXor^=n;

        int lastBit=allXor & ~(allXor-1);

        int n1=0,n2=0;
        for(int n:arr){
            if(n&lastBit) n1^=n;
            else n2^=n;
        }
        return n1>n2 ? (vector<int>{n1,n2}):(vector<int>{n2,n1});
    }
};
```
